#include <bits/stdc++.h>
using namespace std;

int father[1010], isRoot[1010];

int findFather(int x){
    return x == father[x] ? x : findFather(father[x]);
}

void Union(int a, int b){
    int faA = findFather(a);
    int faB = findFather(b);
    if(faA != faB){
        father[faA] = faB; 
    }
}

bool cmp(int a, int b){
    return a > b;
}

void init(int n){
    for(int i = 1; i <= n; i++){
        father[i] = i;
        isRoot[i] = 0;
    }
}

int main()
{
    int n,k,h,course[1010] = {0};
    scanf("%d", &n);
    init(n);
    for(int i = 1; i <= n; i++){
        scanf("%d:", &k);
        for(int j = 0; j < k; j++){
            scanf("%d", &h);
            if(course[h] == 0){
                course[h] = i;
            }
            Union(i, course[h]);
        }
    }
    int ans = 0;
    for(int i = 1; i <= n; i++){
        isRoot[findFather(i)]++;
    }
    for(int i = 1; i <= n; i++){
        if(isRoot[i] != 0){
            ans++;
        }
    }
    sort(isRoot+1,isRoot+n+1,cmp);
    printf("%d\n", ans);
    for(int i = 1; i <= ans; i++){
        printf("%d", isRoot[i]);
        if(i != ans)
            printf(" ");
    }
    return 0;
}
