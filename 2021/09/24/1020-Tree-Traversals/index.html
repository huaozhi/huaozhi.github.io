由后续遍历和中序遍历求层次遍历

#define _CRT_SECURE_NO_WARNINGS
#include <bits/stdc++.h>
using namespace std;

//由后序遍历和中序遍历求层次遍历

vector<int> postOrder, inOrder;
int n;

struct node {
    int data;
    node* left;
    node* right;
    node(int x):data(x),left(nullptr),right(nullptr){}
};

node* createTree(int postL, int postR, int inL, int inR) {
    if (postL > postR)
        return nullptr;
    int data = postOrder[postR];
    node* root = new node(data);
    int index;
    for (index = inL; index < n; index++) {
        if (inOrder[index] == data) {
            break;
        }
    }
    int leftNodes = index - inL;
    root->left = createTree(postL, postL + leftNodes - 1, inL, index - 1);
    root->right = createTree(postL + leftNodes, postR - 1, index + 1, inR);
    return root;
}

int main()
{
    scanf("%d", &n);
    postOrder.resize(n);
    inOrder.resize(n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &postOrder[i]);
    }
    for (int i = 0; i < n; i++) {
        scanf("%d", &inOrder[i]);
    }
    node* root = createTree(0, n - 1, 0, n - 1);
    queue<node*> q;
    q.push(root);
    int flag = 1;
    while (!q.empty()) {
        node* nd = q.front();
        q.pop();
        if (flag == 1) {
            flag = 0;
        }
        else {
            printf(" ");
        }
        printf("%d", nd->data);
        if (nd->left != nullptr) {
            q.push(nd->left);
        }
        if (nd->right != nullptr) {
            q.push(nd->right);
        }
    }
    return 0;
}
