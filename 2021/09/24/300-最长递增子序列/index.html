300.最长递增子序列

2021-09-24 力扣 188 字
动态规划

我的代码

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        vector<int> dp(n,1);
        for(int i = 1; i < n; i++){
            int max = 0;
            for(int j = 0; j < i; j++){
                if(nums[j] < nums[i]){
                    max =  dp[j] > max ? dp[j] : max;
                }
            }
            dp[i] = max+1;
        }
        int max = dp[0];
        for(int i = 1; i < n; i++){
            if(max < dp[i])
            {
                max = dp[i];
            }
        }
        return max;
    }
};

官方题解

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = (int)nums.size();
        if (n == 0) {
            return 0;
        }
        vector<int> dp(n, 0);
        for (int i = 0; i < n; ++i) {
            dp[i] = 1;
            for (int j = 0; j < i; ++j) {
                if (nums[j] < nums[i]) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
        }
        return *max_element(dp.begin(), dp.end());
    }
};
